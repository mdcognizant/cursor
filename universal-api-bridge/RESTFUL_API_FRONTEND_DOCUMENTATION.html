üêç Python Examples

import requests import asyncio import aiohttp # Simple synchronous request def simple_request(): url = "http://localhost:8000/api/v1/process" headers = { "Authorization": "Bearer YOUR_API_KEY", "Content-Type": "application/json" } data = { "data": {"message": "Hello from Python!"}, "service": "echo-service" } response = requests.post(url, json=data, headers=headers) result = response.json() print(f"Success: {result['success']}") print(f"Processing time: {result['processing_time_ms']}ms") print(f"Result: {result['result']}") # Asynchronous batch processing async def batch_processing(): async with aiohttp.ClientSession() as session: url = "http://localhost:8000/api/v1/batch" headers = { "Authorization": "Bearer YOUR_API_KEY", "Content-Type": "application/json" } batch_data = { "requests": [ {"id": f"req_{i}", "data": {"number": i}} for i in range(100) ], "options": {"parallel_processing": True} } async with session.post(url, json=batch_data, headers=headers) as response: result = await response.json() print(f"Processed {result['total_requests']} requests") print(f"Success rate: {result['successful_requests']/result['total_requests']*100:.1f}%") # Performance monitoring def monitor_performance(): response = requests.get("http://localhost:8000/metrics") metrics = response.json() print(f"Current QPS: {metrics['requests_per_second']}") print(f"Average latency: {metrics['latency']['avg_ms']}ms") print(f"Success rate: {metrics['success_rate']}%") print(f"Cache hit rate: {metrics['optimization']['cache_hit_rate']}%") if __name__ == "__main__": simple_request() asyncio.run(batch_processing()) monitor_performance()

üõ†Ô∏è Error Handling Best Practices

# Python error handling example import requests import time def robust_api_call(data, max_retries=3, backoff_factor=1.5): url = "http://localhost:8000/api/v1/process" headers = { "Authorization": "Bearer YOUR_API_KEY", "Content-Type": "application/json" } for attempt in range(max_retries): try: response = requests.post(url, json=data, headers=headers, timeout=30) if response.status_code == 200: return response.json() elif response.status_code == 429: # Rate limit retry_after = int(response.headers.get('Retry-After', 60)) print(f"Rate limited. Waiting {retry_after} seconds...") time.sleep(retry_after) continue elif response.status_code in [500, 502, 503]: # Server errors if attempt < max_retries - 1: wait_time = backoff_factor ** attempt print(f"Server error. Retrying in {wait_time} seconds...") time.sleep(wait_time) continue else: raise Exception(f"Server error after {max_retries} attempts") else: # Client errors (400, 401, 403, 404) error_info = response.json() raise Exception(f"Client error: {error_info['error']['message']}") except requests.exceptions.Timeout: if attempt < max_retries - 1: print("Request timeout. Retrying...") continue else: raise Exception("Request timeout after retries") except requests.exceptions.ConnectionError: if attempt < max_retries - 1: print("Connection error. Retrying...") time.sleep(backoff_factor ** attempt) continue else: raise Exception("Connection error after retries") raise Exception("Max retries exceeded") # Usage try: result = robust_api_call({"data": {"message": "Hello!"}}) print("Success:", result) except Exception as e: print("Error:", str(e))
